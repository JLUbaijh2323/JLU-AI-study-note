void Partitiaon(R[], int m){
	int l = m + 1, r =R.size();
	int k =R[m];
	while(l < r){
		while(l <= r && R[l] >= k){
			l++;
		}
		while(R[r] < k){
			r--;
		}
		swap(R[m], R[k]);
	}
}
void swap(int a, int b){
	int t = a;
	a = b;
	b = t; 
}
